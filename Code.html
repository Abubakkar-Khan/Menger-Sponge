MENGER SPONGE CODE


            class Box {
        constructor(x, y, z, r) {
            this.pos = createVector(x, y, z);
            this.r = r;
        }

        show() {
            push();
            translate(this.pos.x, this.pos.y, this.pos.z);
            fill(255, 50, 50);
            box(this.r);
            pop();
        }

        generate() {
            let boxes = [];
            let newR = this.r / 3;

            for (let x = -1; x < 2; x++) {
                for (let y = -1; y < 2; y++) {
                    for (let z = -1; z < 2; z++) {
                        if (x === 0 && y === 0 && z === 0) continue;
                        let sum = abs(x) + abs(y) + abs(z);
                        if (!(sum <= 1)) {
                            let b = new Box(
                                this.pos.x + x * newR,
                                this.pos.y + y * newR,
                                this.pos.z + z * newR,
                                newR
                                );
                                boxes.push(b);
                            }
                        }
                }
            }
            
            return boxes;
        }
    }

    // Main setup code
    let sponge = [];
    function setup() {
        createCanvas(500, 500, WEBGL);
        sponge.push(new Box(0, 0, 0, 200));
    }
    
    function draw() {
        background(22);
        rotateX(frameCount * 0.01);
        rotateY(frameCount * 0.01);
        for (let b of sponge) {
            b.show();
        }
    }
